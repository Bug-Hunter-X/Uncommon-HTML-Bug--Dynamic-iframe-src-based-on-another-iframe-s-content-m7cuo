The solution addresses the asynchronous loading issue by waiting for the first iframe to fully load before attempting to access its content and setting the second iframe's source. This ensures that the data is available before it is used.

<iframe id="iframe1" src="iframe1.html"></iframe>
<iframe id="iframe2"></iframe>
<script>
  const iframe1 = document.getElementById('iframe1');
  iframe1.onload = function() {
    const iframe1Content = iframe1.contentWindow.document.body.textContent;
    const iframe2 = document.getElementById('iframe2');
    const src = iframe1Content.trim() === 'valid' ? 'iframe2valid.html' : 'iframe2invalid.html';
    iframe2.src = src;
  };
</script>
